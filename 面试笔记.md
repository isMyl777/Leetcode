#2、数组中找出A=B+2C  
```python
n = [1,2,3,4,5,6,7,8]
n.sort(reverse= True)
print(n)
count = 0
for i in range(len(n)):
    target = i
    tem_n = n[i+1:]
    tem = [x*2 for x in tem_n]
    for i1 in tem:
        res = target - i1
        if res in tem_n:
            count += 1
            if res != int(i1/2):
                print(target, int(i1/2), res)
if count == 0:
    print(0)  
```

```python
import random
n = 0
s = ''
while n < 6:
    key = random.choice(range(0, 4));
    if key == 1:
        s += str(random.randint(0,9))
    elif key == 0:
        s += chr(random.randint(97, 122))
    else:
        s += chr(random.randint(65, 90))
    n+=1
print(s)
```
2、输入字符串，返回第一个只出现一次字母，如果没有返回空  
```python
def onlyOnce(str1):
    res = ''
    for i in str1:
        if str1.count(i) == 1:
            res = i
            break
    else:
            res = ' '
    return res
s = input()
print(onlyOnce(s))
```
